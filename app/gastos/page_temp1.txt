"use client";

import { useState, useEffect, useMemo, useCallback } from "react";
import { Plus, Lightbulb, Droplets, Calendar, DollarSign, TrendingUp, ChevronDown, ChevronUp, X, FileText } from "lucide-react";
import DashboardLayout from "@/components/DashboardLayout";
import ModalNuevoGasto from "./ModalNuevoGasto";

export interface Local {
  _id: string;
  nombre: string;
  tipo: 'panaderia' | 'spa' | 'profesor' | 'casa';
  direccion: string;
}

interface Lectura {
  localId: Local;
  medidorNumero?: number;
  lecturaAnterior: number;
  lecturaActual: number;
  consumo: number;
}

interface Gasto {
  _id: string;
  mes: string;
  tipo: 'luz' | 'agua';
  consumoTotal: number;
  montoTotal: number;
  cargoFijo?: number;
  igv?: number;
  otrosCargos?: number;
  lecturas: Lectura[];
  costosPorLocal: {
    localId: Local;
    consumo: number;
    monto: number;
  }[];
  createdAt: string;
}

export default function GastosPage() {
  const [gastos, setGastos] = useState<Gasto[]>([]);
  const [locales, setLocales] = useState<Local[]>([]);
  const [loading, setLoading] = useState(true);
  const [showNuevoGasto, setShowNuevoGasto] = useState(false);
  const [filtroTipo, setFiltroTipo] = useState<'todos' | 'luz' | 'agua'>('todos');
  const [filtroMes, setFiltroMes] = useState<string>('');
  const [expandedGasto, setExpandedGasto] = useState<string | null>(null);

  // Form state
  const [mes, setMes] = useState('');
  const [tipo, setTipo] = useState<'luz' | 'agua'>('luz');
  const [consumoTotal, setConsumoTotal] = useState('');
  const [montoTotal, setMontoTotal] = useState('');
  const [cargoFijo, setCargoFijo] = useState('');
  const [igv, setIgv] = useState('');
  const [otrosCargos, setOtrosCargos] = useState('');
  const [lecturas, setLecturas] = useState<{
    localId: string;
    medidorNumero?: number;
    lecturaAnterior: number;
    lecturaActual: number;
  }[]>([]);

  useEffect(() => {
    cargarDatos();
  }, []);

  const cargarDatos = async () => {
    try {
      const [gastosRes, localesRes] = await Promise.all([
        fetch('/api/gastos'),
        fetch('/api/locales')
      ]);

      const gastosData = await gastosRes.json();
      const localesData = await localesRes.json();

      if (gastosData.success) {
        setGastos(gastosData.gastos);
      }

      if (localesData.success) {
        setLocales(localesData.locales);
        // Inicializar lecturas solo con los locales que NO son casa
        const localesConMedidor = localesData.locales.filter((l: Local) => l.tipo !== 'casa');
        setLecturas(localesConMedidor.map((local: Local) => ({
          localId: local._id,
          medidorNumero: 1,
          lecturaAnterior: 0,
          lecturaActual: 0,
        })));
      }

      setLoading(false);
    } catch (error) {
      console.error('Error al cargar datos:', error);
      setLoading(false);
    }
  };

  const inicializarLocales = async () => {
    try {
      const res = await fetch('/api/locales/init', { method: 'POST' });
      const data = await res.json();

      if (data.success) {
        alert('Locales inicializados correctamente');
        cargarDatos();
      } else {
        alert(data.message);
      }
    } catch (error) {
      console.error('Error:', error);
      alert('Error al inicializar locales');
    }
  };

  const handleSubmitGasto = async (e: React.FormEvent) => {
    e.preventDefault();

    if (!mes || !consumoTotal || !montoTotal || lecturas.length === 0) {
      alert('Por favor completa todos los campos obligatorios');
      return;
    }

    try {
      const res = await fetch('/api/gastos', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          mes,
          tipo,
          consumoTotal: parseFloat(consumoTotal),
          montoTotal: parseFloat(montoTotal),
          cargoFijo: cargoFijo ? parseFloat(cargoFijo) : 0,
          igv: igv ? parseFloat(igv) : 0,
          otrosCargos: otrosCargos ? parseFloat(otrosCargos) : 0,
          lecturas,
        }),
      });

      const data = await res.json();

      if (data.success) {
        alert('Gasto registrado exitosamente');
        setShowNuevoGasto(false);
        resetForm();
        cargarDatos();
      } else {
        alert(data.message || 'Error al registrar gasto');
      }
    } catch (error) {
      console.error('Error:', error);
      alert('Error al registrar gasto');
    }
  };

  const resetForm = () => {
    setMes('');
    setTipo('luz');
    setConsumoTotal('');
    setMontoTotal('');
    setCargoFijo('');
    setIgv('');
    setOtrosCargos('');
    const localesConMedidor = locales.filter(l => l.tipo !== 'casa');
    setLecturas(localesConMedidor.map(local => ({
      localId: local._id,
      medidorNumero: 1,
      lecturaAnterior: 0,
      lecturaActual: 0,
    })));
  };

  const updateLectura = useCallback((index: number, field: string, value: number) => {
    setLecturas(prev => {
      const newLecturas = [...prev];
      (newLecturas[index] as any)[field] = value;
      return newLecturas;
    });
  }, []);

  // 1. Función de filtrado
  const gastosFiltr = gastos.filter(gasto => {
    const coincideTipo = filtroTipo === 'todos' || gasto.tipo === filtroTipo;
    const coincideMes = !filtroMes || gasto.mes === filtroMes;
    return coincideTipo && coincideMes;
  });

  // 2. Función para calcular costo por unidad
  const calcularCostoPorUnidad = (gasto: Gasto): number => {
    if (gasto.consumoTotal === 0) return 0;
    return gasto.montoTotal / gasto.consumoTotal;
  };

  // Calcular consumo de lecturas (locales con medidor)
  const calcularConsumoLecturas = (): number => {
    return lecturas.reduce((total, lectura) => {
      const consumo = lectura.lecturaActual - lectura.lecturaAnterior;
      return total + (consumo > 0 ? consumo : 0);
    }, 0);
  };

  // Calcular consumo de casa (automático)
  const calcularConsumoCasa = (): number => {
    const totalConsumo = parseFloat(consumoTotal) || 0;
    const consumoLecturas = calcularConsumoLecturas();
    const consumoCasa = totalConsumo - consumoLecturas;
    return consumoCasa > 0 ? consumoCasa : 0;
  };

  // Estados para navegación del modal
  const [pasoModal, setPasoModal] = useState(1);

  // Memoizar cálculos para uso en el modal
  const consumoLecturas = useMemo(() => calcularConsumoLecturas(), [lecturas]);
  const consumoCasa = useMemo(() => calcularConsumoCasa(), [consumoTotal, lecturas]);

  // Obtener meses únicos para filtro
  const mesesUnicos = Array.from(new Set(gastos.map(g => g.mes))).sort().reverse();

  // 3. Componente de tarjeta de gasto expandible
  const GastoCard = ({ gasto }: { gasto: Gasto }) => {
    const isExpanded = expandedGasto === gasto._id;
    const costoPorUnidad = calcularCostoPorUnidad(gasto);
    const localCasa = gasto.lecturas.find(l => l.localId.tipo === 'casa');

    const colores = gasto.tipo === 'luz'
      ? { bg: 'bg-white', border: 'border-gray-100', text: 'text-[#0A2640]', icon: 'bg-[#0A2640]/10', iconColor: 'text-[#0A2640]' }
      : { bg: 'bg-white', border: 'border-gray-100', text: 'text-[#0A2640]', icon: 'bg-[#0A2640]/10', iconColor: 'text-[#0A2640]' };

    return (
      <div className={`${colores.bg} rounded-2xl shadow-sm border ${colores.border} overflow-hidden transition-all duration-300 hover:shadow-md`}>
        {/* Header */}
        <div className="p-6">
          <div className="flex items-start justify-between mb-4">
            <div className="flex items-center gap-3">
              <div className={`${colores.icon} p-3 rounded-lg`}>
                {gasto.tipo === 'luz' ? (
                  <Lightbulb className={`w-6 h-6 ${colores.iconColor}`} />
                ) : (
                  <Droplets className={`w-6 h-6 ${colores.iconColor}`} />
                )}
              </div>
              <div>
                <h3 className="text-lg font-semibold text-gray-800">{gasto.mes}</h3>
                <p className="text-sm text-gray-500 capitalize">{gasto.tipo}</p>
              </div>
            </div>
            <button
              onClick={() => setExpandedGasto(isExpanded ? null : gasto._id)}
              className={`${colores.icon} p-2 rounded-lg transition-transform duration-200 ${isExpanded ? 'rotate-180' : ''}`}
            >
              <ChevronDown className={`w-5 h-5 ${colores.iconColor}`} />
            </button>
          </div>

          <div className="grid grid-cols-2 gap-4">
            <div className="bg-gray-50 rounded-lg p-3">
              <p className="text-xs text-gray-500 mb-1">Consumo Total</p>
              <p className={`text-xl font-bold ${colores.text}`}>
                {gasto.consumoTotal.toFixed(2)} {gasto.tipo === 'luz' ? 'kWh' : 'm³'}
              </p>
            </div>
            <div className="bg-gray-50 rounded-lg p-3">
              <p className="text-xs text-gray-500 mb-1">Monto Total</p>
              <p className={`text-xl font-bold ${colores.text}`}>
                S/ {gasto.montoTotal.toFixed(2)}
              </p>
            </div>
          </div>
        </div>

        {/* Contenido expandido */}
        {isExpanded && (
          <div className="border-t border-gray-100 p-6 space-y-6">
            {/* Desglose de costos */}
            <div className="space-y-3">
              <h4 className="text-sm font-semibold text-gray-800 flex items-center gap-2">
                <DollarSign className="w-4 h-4" />
                Desglose de Costos
              </h4>
              <div className="bg-gray-50 rounded-lg p-4 space-y-2">
                {gasto.cargoFijo && gasto.cargoFijo > 0 && (
                  <div className="flex justify-between text-sm">
                    <span className="text-gray-500">Cargo Fijo:</span>
                    <span className="text-gray-800">S/ {gasto.cargoFijo.toFixed(2)}</span>
                  </div>
                )}
                {gasto.igv && gasto.igv > 0 && (
                  <div className="flex justify-between text-sm">
                    <span className="text-gray-500">IGV:</span>
                    <span className="text-gray-800">S/ {gasto.igv.toFixed(2)}</span>
                  </div>
                )}
                {gasto.otrosCargos && gasto.otrosCargos > 0 && (
                  <div className="flex justify-between text-sm">
                    <span className="text-gray-500">Otros Cargos:</span>
                    <span className="text-gray-800">S/ {gasto.otrosCargos.toFixed(2)}</span>
                  </div>
                )}
                <div className={`flex justify-between text-sm ${(gasto.cargoFijo || gasto.igv || gasto.otrosCargos) ? 'pt-2 border-t border-gray-200' : ''} font-semibold`}>
                  <span className={colores.text}>Costo por {gasto.tipo === 'luz' ? 'kWh' : 'm³'}:</span>
                  <span className={colores.text}>S/ {costoPorUnidad.toFixed(4)}</span>
                </div>
              </div>
            </div>

            {/* Lecturas por local - Solo locales con medidor, NO casa */}
            <div className="space-y-3">
              <h4 className="text-sm font-semibold text-gray-800 flex items-center gap-2">
                <TrendingUp className="w-4 h-4" />
                Lecturas de Medidores
              </h4>
              <div className="space-y-2">
                {gasto.lecturas.filter(l => l.localId.tipo !== 'casa').map((lectura, idx) => (
                  <div
                    key={idx}
                    className="bg-gray-50 rounded-lg p-3 flex items-center justify-between"
                  >
                    <div className="flex items-center gap-3">
                      <div>
                        <div className="flex items-center gap-2">
                          <p className="text-sm font-medium text-gray-800">{lectura.localId.nombre}</p>

                        </div>
                        <p className="text-xs text-gray-500">
                          {lectura.lecturaAnterior} → {lectura.lecturaActual}
                        </p>
                      </div>
                    </div>
                    <div className="text-right">
                      <p className={`text-lg font-bold ${colores.text}`}>{lectura.consumo.toFixed(2)}</p>
                      <p className="text-xs text-gray-500">{gasto.tipo === 'luz' ? 'kWh' : 'm³'}</p>
                    </div>

                ))}
              </div>
            </div>

            {/* Costos por local - Solo locales a cobrar, NO casa */}
            <div className="space-y-3">
              <div className="flex items-center justify-between">
                <h4 className="text-sm font-semibold text-gray-800 flex items-center gap-2">
                  <DollarSign className="w-4 h-4" />

                </h4>

                  onClick={() => generarReporte(gasto)}

                  Generar Reporte

                  <div

    const costoPorUnidad = calcularCostoPorUnidad(gasto);

      white: '#FFFFFF',
      text: '#0A2640',
      textMedium: '#64748b',
      textLight: '#94a3b8',
      success: '#059669',
      divider: '#CBD5E1'

    // Fondo blanco limpio
    ctx.fillStyle = colors.white;
    ctx.fillRect(0, 0, width, height);

    ctx.fillStyle = colors.primary;

    ctx.font = '13px Arial';

    ctx.fillText(`Fecha: ${fecha}`, width - 60, yPos + 28);

    yPos += 70;

    ctx.font = 'bold 64px Arial';
    ctx.fillStyle = colors.primary;
    const icon = gasto.tipo === 'luz' ? '⚡' : '💧';
    ctx.fillText(icon, 80, yPos);

    ctx.fillText(gasto.tipo === 'luz' ? 'Luz' : 'Agua', 170, yPos);

    ctx.font = '22px Arial';
    ctx.fillStyle = colors.textMedium;
    ctx.fillText(gasto.mes, 170, yPos + 40);

    yPos += 80;

    // Línea divisoria
    ctx.strokeStyle = colors.softGray;
    ctx.lineWidth = 1;
    ctx.beginPath();

    ctx.lineTo(width - 80, yPos);
    ctx.stroke();

    yPos += 50;

    ctx.fillStyle = colors.text;
    ctx.fillText('Resumen Financiero', 80, yPos);

    yPos += 45;

    // Datos con fondos alternados
    const datos: Array<{ label: string; value: string; highlight?: boolean }> = [

  // Función auxiliar para dibujar rectángulos con bordes redondeados
  function roundRect(ctx: CanvasRenderingContext2D, x: number, y: number, width: number, height: number, radius: number) {
    ctx.beginPath();
    ctx.moveTo(x + radius, y);
    ctx.lineTo(x + width - radius, y);
    ctx.quadraticCurveTo(x + width, y, x + width, y + radius);
    ctx.lineTo(x + width, y + height - radius);
    ctx.quadraticCurveTo(x + width, y + height, x + width - radius, y + height);
    ctx.lineTo(x + radius, y + height);
    ctx.quadraticCurveTo(x, y + height, x, y + height - radius);
    ctx.lineTo(x, y + radius);
    ctx.quadraticCurveTo(x, y, x + radius, y);
    ctx.closePath();
  }
